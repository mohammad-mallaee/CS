اگر 
$U \subseteq \mathscr{F}$
اثبات کنید:
\begin{enumerate}
    \item اگر
    $U$
    صدق پذیر باشد و 
    $A \in U$
    آنگاه
    $U - \{A\}$
    نیز صدق پذیر است.

    \item اگر
    $U$
    صدق پذیر باشد و
    $B$
    معتبر باشد آنگاه
    $U \cup \{B\}$
    نیز صدق پذیر است.

    \item اگر
    $U$
    صدق ناپذیر باشد آنگاه برای هر فرمول
    $B$
    مجموعه
    $U \cup {B}$
    نیز صدق ناپذیر است.

    \item اگر
    $U$
    ناپذیر باشد و فرمول
    $A \in U$
    معتبر باشد
    آنگاه 
    $U - \{A\}$
    نیز صدق ناپذیر است.
\end{enumerate}
